#ifndef PX_TRIANGLE_MESH_ANALYSIS_RESULT_H
#define PX_TRIANGLE_MESH_ANALYSIS_RESULT_H


#include "PxPhysXConfig.h"
#include "foundation/PxFlags.h"

#if !PX_DOXYGEN
namespace physx
{
#endif

    class PxTriangleMeshAnalysisResult
    {
    public:
        enum Enum
        {
            eVALID = 0,
            eZERO_VOLUME = (1 << 0),
            eOPEN_BOUNDARIES = (1 << 1),
            eSELF_INTERSECTIONS = (1 << 2),
            eINCONSISTENT_TRIANGLE_ORIENTATION = (1 << 3),
            eCONTAINS_ACUTE_ANGLED_TRIANGLES = (1 << 4),
            eEDGE_SHARED_BY_MORE_THAN_TWO_TRIANGLES = (1 << 5),
            eCONTAINS_DUPLICATE_POINTS = (1 << 6),
            eCONTAINS_INVALID_POINTS = (1 << 7),
            eREQUIRES_32BIT_INDEX_BUFFER = (1 << 8),

            eMESH_IS_PROBLEMATIC = (1 << 9),
            eMESH_IS_INVALID = (1 << 10)
        };
    };
    typedef PxFlags<PxTriangleMeshAnalysisResult::Enum, PxU32> PxTriangleMeshAnalysisResults;
    PX_FLAGS_OPERATORS(PxTriangleMeshAnalysisResult::Enum, PxU32)

#if !PX_DOXYGEN
}
