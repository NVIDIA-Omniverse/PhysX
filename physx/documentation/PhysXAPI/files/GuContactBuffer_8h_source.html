70  PX_FORCE_INLINE bool contact(const PxVec3& worldPoint,

73  PxU32 faceIndex1 = PXC_CONTACT_NO_FACE_INDEX

77 
78  if(count>=MAX_CONTACTS)
79  return false;
80 
81  Gu::ContactPoint& p = contacts[count++];
82  p.normal = worldNormalIn;
83  p.point = worldPoint;
84  p.separation = separation;
85  p.internalFaceIndex1= faceIndex1;
86  return true;
87  }
88 
89  PX_FORCE_INLINE bool contact(const Gu::ContactPoint & pt)
90  {
91  if(count>=MAX_CONTACTS)
92  return false;
93  contacts[count++] = pt;
94  return true;
95  }
96 
97  PX_FORCE_INLINE Gu::ContactPoint* contact()
98  {
99  if(count>=MAX_CONTACTS)
100  return NULL;
101  return &contacts[count++];
