60  void* allocate(size_t size, const char* file, int line)
61  {
62  size_t pad = N - 1 + sizeof(size_t); // store offset for delete.
63  uint8_t* base = reinterpret_cast<uint8_t*>(BaseAllocator::allocate(size + pad, file, line));
64  if (!base)
65  return NULL;
66 
67  uint8_t* ptr = reinterpret_cast<uint8_t*>(size_t(base + pad) & ~(size_t(N) - 1)); // aligned pointer, ensuring N
68  // is a size_t
69  // wide mask
70  reinterpret_cast<size_t*>(ptr)[-1] = size_t(ptr - base); // store offset
71 
72  return ptr;
73  }
74  void deallocate(void* ptr)
75  {
76  if (ptr == NULL)
77  return;
78 
79  uint8_t* base = reinterpret_cast<uint8_t*>(ptr) - reinterpret_cast<size_t*>(ptr)[-1];
80  BaseAllocator::deallocate(base);
81  }
82  };
