61 // PT: the rules are simple:
62 // - PX_ALLOC/PX_ALLOCATE/PX_FREE is similar to malloc/free. Use that for POD/anything that doesn't need ctor/dtor.
63 // - PX_NEW/PX_DELETE is similar to new/delete. Use that for anything that needs a ctor/dtor.
64 // - Everything goes through the user allocator.
65 // - Inherit from PxUserAllocated to PX_NEW something. Do it even on small classes, it's free.
66 // - You cannot PX_NEW a POD. Use PX_ALLOC.
67 
68 #define PX_ALLOC(n, name) physx::PxAllocator().allocate(n, __FILE__, __LINE__)

70 // PT: use this one to reduce the amount of visible reinterpret_cast
71 #define PX_ALLOCATE(type, count, name) reinterpret_cast<type*>(PX_ALLOC(count*sizeof(type), name))

149  /*
150  \brief Virtual allocator to be used by foundation types to provide run-time defined allocators.
151  Due to the fact that Array extends its allocator, rather than contains a reference/pointer to it, the VirtualAllocator
152  must
153  be a concrete type containing a pointer to a virtual callback. The callback may not be available at instantiation time,
154  therefore
155  methods are provided to set the callback later.
156  */
157  class PxVirtualAllocator
158  {
159  public:
160  PxVirtualAllocator(PxVirtualAllocatorCallback* callback = NULL, const int group = 0) : mCallback(callback), mGroup(group) {}
161 
162  PX_FORCE_INLINE void* allocate(const size_t size, const char* file, const int line)
163  {
164  PX_ASSERT(mCallback);
165  if (size)
166  return mCallback->allocate(size, mGroup, file, line);
167  return NULL;
168  }
169 
170  PX_FORCE_INLINE void deallocate(void* ptr)
171  {
172  PX_ASSERT(mCallback);
173  if (ptr)
174  mCallback->deallocate(ptr);
175  }
176 
177  private:
178  PxVirtualAllocatorCallback* mCallback;
179  const int mGroup;
180  PxVirtualAllocator& operator=(const PxVirtualAllocator&);
181  };
