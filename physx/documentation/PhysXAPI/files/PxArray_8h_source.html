410  PX_INLINE void removeRange(uint32_t begin, uint32_t count)
412  {
413  PX_ASSERT(begin < mSize);
414  PX_ASSERT((begin + count) <= mSize);
415 
416  for(uint32_t i = 0; i < count; i++)
417  mData[begin + i].~T(); // call the destructor on the ones being removed first.
418 
419  T* dest = &mData[begin]; // location we are copying the tail end objects to
420  T* src = &mData[begin + count]; // start of tail objects
421  uint32_t move_count = mSize - (begin + count); // compute remainder that needs to be copied down
422 
423  for(uint32_t i = 0; i < move_count; i++)
424  {
425  PX_PLACEMENT_NEW(dest, T(*src)); // copy the old one to the new location
426  src->~T(); // call the destructor on the old location
427  dest++;
428  src++;
429  }
430  mSize -= count;
431  }

682 template <class T, class Alloc>
683 PX_NOINLINE T& PxArray<T, Alloc>::growAndPushBack(const T& a)
684 {
685  uint32_t capacity = capacityIncrement();
686 
687  T* newData = allocate(capacity);
688  PX_ASSERT((!capacity) || (newData && (newData != mData)));
689  copy(newData, newData + mSize, mData);
690 
691  // inserting element before destroying old array
692  // avoids referencing destroyed object when duplicating array element.
693  PX_PLACEMENT_NEW(reinterpret_cast<void*>(newData + mSize), T)(a);
694 
695  destroy(mData, mData + mSize);
696  if(!isInUserMemory())
697  deallocate(mData);
698 
699  mData = newData;
700  mCapacity = capacity;
701 
702  return mData[mSize++];
703 }
