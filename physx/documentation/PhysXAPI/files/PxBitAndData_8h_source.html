42 template <typename storageType, storageType bitMask>
43 class PxBitAndDataT
44 {
45  public:
46  PX_FORCE_INLINE PxBitAndDataT(const PxEMPTY)
47  {
48  }
49  PX_FORCE_INLINE PxBitAndDataT() : mData(0)
50  {
51  }
52  PX_FORCE_INLINE PxBitAndDataT(storageType data, bool bit = false)
53  {
54  mData = bit ? storageType(data | bitMask) : data;
55  }
56 
57  PX_CUDA_CALLABLE PX_FORCE_INLINE operator storageType() const
58  {
59  return storageType(mData & ~bitMask);
60  }
61  PX_CUDA_CALLABLE PX_FORCE_INLINE void setBit()
62  {
63  mData |= bitMask;
64  }
65  PX_CUDA_CALLABLE PX_FORCE_INLINE void clearBit()
66  {
67  mData &= ~bitMask;
68  }
69  PX_CUDA_CALLABLE PX_FORCE_INLINE storageType isBitSet() const
70  {
71  return storageType(mData & bitMask);
72  }
73 
74  protected:
75  storageType mData;
76 };
