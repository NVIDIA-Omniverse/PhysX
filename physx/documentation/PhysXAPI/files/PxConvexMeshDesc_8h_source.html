255 PX_INLINE bool PxConvexMeshDesc::isValid() const
256 {
257  // Check geometry
258  if(points.count < 3 || //at least 1 trig's worth of points
259  (points.count > 0xffff && flags & PxConvexFlag::e16_BIT_INDICES))
260  return false;
261  if(!points.data)
262  return false;
263  if(points.stride < sizeof(PxVec3)) //should be at least one point's worth of data
264  return false;
265  if (quantizedCount < 4)
266  return false;
267 
268  // Check topology
269  if(polygons.data)
270  {
271  if(polygons.count < 4) // we require 2 neighbors for each vertex - 4 polygons at least
272  return false;
273 
274  if(!indices.data) // indices must be provided together with polygons
275  return false;
276 
277  PxU32 limit = (flags & PxConvexFlag::e16_BIT_INDICES) ? sizeof(PxU16) : sizeof(PxU32);
278  if(indices.stride < limit)
279  return false;
280 
281  limit = sizeof(PxHullPolygon);
282  if(polygons.stride < limit)
283  return false;
284  }
285  else
286  {
287  // We can compute the hull from the vertices
288  if(!(flags & PxConvexFlag::eCOMPUTE_CONVEX))
289  return false; // If the mesh is convex and we're not allowed to compute the hull,
290  // you have to provide it completely (geometry & topology).
291  }
292 
293  if((flags & PxConvexFlag::ePLANE_SHIFTING) && vertexLimit < 4)
294  {
295  return false;
296  }
297 
298  if (!(flags & PxConvexFlag::ePLANE_SHIFTING) && vertexLimit < 8)
299  {
300  return false;
301  }
302 
303  if(vertexLimit > 256)
304  {
305  return false;
306  }
307 
308  if (sdfDesc && !sdfDesc->isValid())
309  {
310  return false;
311  }
312 
313  return true;
314 }
