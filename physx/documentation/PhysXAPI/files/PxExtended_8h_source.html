138  PX_INLINE void cross(const PxExtendedVec3& left, const PxVec3& right)
139  {
140  // temps needed in case left or right is this.
141  PxExtended a = (left.y * PxExtended(right.z)) - (left.z * PxExtended(right.y));
142  PxExtended b = (left.z * PxExtended(right.x)) - (left.x * PxExtended(right.z));
143  PxExtended c = (left.x * PxExtended(right.y)) - (left.y * PxExtended(right.x));
