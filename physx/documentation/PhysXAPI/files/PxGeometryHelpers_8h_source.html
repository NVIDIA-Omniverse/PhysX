190  PX_FORCE_INLINE void storeAny(const PxGeometry& geometry)
191  {
192  PX_ASSERT_WITH_MESSAGE( (geometry.getType() >= PxGeometryType::eSPHERE) &&
193  (geometry.getType() < PxGeometryType::eGEOMETRY_COUNT),
194  "Unexpected GeometryType in PxGeometryHolder::storeAny");
195 
196  switch(geometry.getType())
197  {
198  case PxGeometryType::eSPHERE: put<PxSphereGeometry>(geometry); break;
199  case PxGeometryType::ePLANE: put<PxPlaneGeometry>(geometry); break;
200  case PxGeometryType::eCAPSULE: put<PxCapsuleGeometry>(geometry); break;
201  case PxGeometryType::eBOX: put<PxBoxGeometry>(geometry); break;
202  case PxGeometryType::eCONVEXMESH: put<PxConvexMeshGeometry>(geometry); break;
203  case PxGeometryType::eTRIANGLEMESH: put<PxTriangleMeshGeometry>(geometry); break;
204  case PxGeometryType::eTETRAHEDRONMESH: put<PxTetrahedronMeshGeometry>(geometry); break;
205  case PxGeometryType::eHEIGHTFIELD: put<PxHeightFieldGeometry>(geometry); break;
206  case PxGeometryType::ePARTICLESYSTEM: put<PxParticleSystemGeometry>(geometry); break;
207  case PxGeometryType::eHAIRSYSTEM: put<PxHairSystemGeometry>(geometry); break;
208  case PxGeometryType::eCUSTOM: put<PxCustomGeometry>(geometry); break;
209  case PxGeometryType::eGEOMETRY_COUNT:
210  case PxGeometryType::eINVALID: break;
211  }
212  }
