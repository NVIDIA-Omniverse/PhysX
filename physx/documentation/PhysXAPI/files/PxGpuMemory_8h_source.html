193  void initialize(PxCudaContextManager* cudaContextManager)
194  {
195  mCudaContextManager = cudaContextManager;
196  }
197 
198  //If false is returned, then the memory must be freed first
199  bool allocate(PxI32 numElements, bool autoReleaseExistingBuffer = true)
200  {
201  if (mPointer)
202  {
203  if (!autoReleaseExistingBuffer)
204  return false;
205  release();
206  }
207 
208  PxGpuMemory::allocDeviceBuffer(mCudaContextManager, mPointer, numElements);
209  mNumElements = numElements;
210 
211  return true;
212  }
213 
214  bool release()
215  {
216  if (!mPointer)
217  return false;
218 
219  PxGpuMemory::freeDeviceBuffer(mCudaContextManager, mPointer);
220  mPointer = NULL;
221  mNumElements = 0;
222  return true;
223  }
224
