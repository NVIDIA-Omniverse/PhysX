159  PX_INLINE bool PxHairSystemDesc::isValid() const
160  {
161  if (segmentLength < 0.0f || segmentRadius < 0.0f)
162  return false;
163 
164  if (2.0f * segmentRadius >= segmentLength)
165  return false;
166 
167  if (numStrands == 0)
168  return false;
169 
170  if (numVerticesPerStrand.count != numStrands)
171  return false;
172 
173  PxU32 totalNumVertices = 0;
174  for (PxU32 i = 0; i < numVerticesPerStrand.count; i++)
175  {
176  const PxU32 numVertices = numVerticesPerStrand.at<PxU32>(i);
177  totalNumVertices += numVertices;
178  if (numVertices < 2)
179  {
180  return false;
181  }
182  }
183 
184  if (vertices.count != totalNumVertices && vertices.count != numStrands)
185  return false;
186 
187  if (velocities.count != totalNumVertices && velocities.count != 0)
188  return false;
189 
190  return true;
191  }
