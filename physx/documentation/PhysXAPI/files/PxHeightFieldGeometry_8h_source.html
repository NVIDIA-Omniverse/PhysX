144 PX_INLINE bool PxHeightFieldGeometry::isValid() const
145 {
146  if(mType != PxGeometryType::eHEIGHTFIELD)
147  return false;
148  if(!PxIsFinite(heightScale) || !PxIsFinite(rowScale) || !PxIsFinite(columnScale))
149  return false;
150  if(rowScale < PX_MIN_HEIGHTFIELD_XZ_SCALE || columnScale < PX_MIN_HEIGHTFIELD_XZ_SCALE || heightScale < PX_MIN_HEIGHTFIELD_Y_SCALE)
151  return false;
152  if(!heightField)
153  return false;
154 
155  return true;
156 }
