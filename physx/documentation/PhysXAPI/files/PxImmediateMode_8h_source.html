292  void initData()
293  {
294  parentPose = PxTransform(PxIdentity);
295  childPose = PxTransform(PxIdentity);
296  frictionCoefficient = 0.05f;
297  maxJointVelocity = 100.0f;
298  type = PxArticulationJointType::eUNDEFINED; // For root
299 
300  for(PxU32 i=0;i<PxArticulationAxis::eCOUNT;i++)
301  {
302  motion[i] = PxArticulationMotion::eLOCKED;
303  limits[i] = PxArticulationLimit(0.0f, 0.0f);
304  drives[i] = PxArticulationDrive(0.0f, 0.0f, 0.0f);
305  armature[i] = 0.0f;
306  jointPos[i] = 0.0f;
307  jointVel[i] = 0.0f;
308  }
309  PxMemSet(targetPos, 0xff, sizeof(PxReal)*PxArticulationAxis::eCOUNT);
310  PxMemSet(targetVel, 0xff, sizeof(PxReal)*PxArticulationAxis::eCOUNT);
311  }
312  };
