106  PX_INLINE PxU64 addGridFilteringPass(PxIsosurfaceGridFilteringType::Enum operation)
107  {
108  for (PxU32 i = 0; i < 32; ++i)
109  {
110  PxIsosurfaceGridFilteringType::Enum o;
111  if (!getGridFilteringPass(i, o))
112  {
113  setGridFilteringPass(i, operation);
114  return i;
115  }
116  }
117  return 0xFFFFFFFFFFFFFFFF;
118  }
119 
126  PX_INLINE void setGridFilteringPass(PxU32 passIndex, PxIsosurfaceGridFilteringType::Enum operation)
127  {
128  PX_ASSERT(passIndex < 32u);
129  PxU32 shift = passIndex * 2;
130  gridFilteringFlags &= ~(PxU64(3) << shift);
131  gridFilteringFlags |= PxU64(operation) << shift;
132  }
133 
141  PX_INLINE bool getGridFilteringPass(PxU32 passIndex, PxIsosurfaceGridFilteringType::Enum& operation) const
142  {
143  PxU32 shift = passIndex * 2;
144  PxU64 v = gridFilteringFlags >> shift;
145  v &= 3; //Extract last 2 bits
146  operation = PxIsosurfaceGridFilteringType::Enum(v);
147  return operation != PxIsosurfaceGridFilteringType::eNONE;
148  }
