287  return PxJointLimitParameters::isValid() &&
288  PxIsFinite(upper) && PxIsFinite(lower) && upper >= lower &&
289  PxIsFinite(upper - lower);
290  }

503  PxJointLimitPyramid(PxReal yLimitAngleMin, PxReal yLimitAngleMax, PxReal zLimitAngleMin, PxReal zLimitAngleMax, PxReal contactDist = -1.0f) :
504  yAngleMin(yLimitAngleMin),
505  yAngleMax(yLimitAngleMax),
506  zAngleMin(zLimitAngleMin),
507  zAngleMax(zLimitAngleMax)
508  {
509  if(contactDist == -1.0f)
510  {
511  const PxReal contactDistY = PxMin(0.1f, 0.49f*(yLimitAngleMax - yLimitAngleMin));
512  const PxReal contactDistZ = PxMin(0.1f, 0.49f*(zLimitAngleMax - zLimitAngleMin));
513  PxJointLimitParameters::contactDistance = contactDist == PxMin(contactDistY, contactDistZ);
514  }
515  else
516  {
517  PxJointLimitParameters::contactDistance = contactDist;
518  }
519 
520  bounceThreshold = 0.5f;
521  }

534  PxJointLimitPyramid(PxReal yLimitAngleMin, PxReal yLimitAngleMax, PxReal zLimitAngleMin, PxReal zLimitAngleMax, const PxSpring& spring) :
535  yAngleMin(yLimitAngleMin),
536  yAngleMax(yLimitAngleMax),
537  zAngleMin(zLimitAngleMin),
538  zAngleMax(zLimitAngleMax)
539  {
540  stiffness = spring.stiffness;
541  damping = spring.damping;
542  }
543 
549  PX_INLINE bool isValid() const
550  {
551  return PxJointLimitParameters::isValid() &&
552  PxIsFinite(yAngleMin) && yAngleMin>-PxPi && yAngleMin<PxPi &&
553  PxIsFinite(yAngleMax) && yAngleMax>-PxPi && yAngleMax<PxPi &&
554  PxIsFinite(zAngleMin) && zAngleMin>-PxPi && zAngleMin<PxPi &&
555  PxIsFinite(zAngleMax) && zAngleMax>-PxPi && zAngleMax<PxPi &&
556  yAngleMax>=yAngleMin && zAngleMax>=zAngleMin;
