106  template<const bool debugExecution>
107  static PX_FORCE_INLINE physx::PxCUresult _testLaunch(physx::PxCUresult resultR, PxKernelWrangler* kernelWrangler, PxCudaContext* ctx, PxU16 id, CUstream hStream, const char* file, PxU32 line)
108  {
109  // PT: don't assert immediately, we want to print the failing kernel's name first
110  if (resultR != CUDA_SUCCESS)
111  outputKernelLaunchError(kernelWrangler, id, file, line);
112  else if (debugExecution)
113  {
114  // PT: launching the kernel might have worked but executing the kernel is a different thing that can also fail.
115  // This is only for debugging so code bloat won't matter here, and we can inline the whole thing.
116  // This assumes the compiler removes all of it, which should work because we use a template for debugExecution.
117  // Note that the launchKernel function itself has a LAUNCH_SYNCHRONOUS mode, so this seems quite redundant.
118  // But that's how the existing code was when I found it, and I assume there must be a reason.
119  const physx::PxCUresult syncErr = ctx->streamSynchronize(hStream);
120  // PT: here again, don't assert immediately
121  if (syncErr != CUDA_SUCCESS)
122  {
123  char buffer[4096];
124  physx::Pxsnprintf(buffer, 4096, "GPU kernel '%s' execution failed!\n", kernelWrangler->getCuFunctionName(id));
125  PxGetFoundation().error(PxErrorCode::eINTERNAL_ERROR, file, line, buffer);
126  }
127  PX_ASSERT(syncErr == CUDA_SUCCESS);
128  }
129  PX_ASSERT(resultR == CUDA_SUCCESS);
130  return resultR;
131  }

182  static void packArguments(PxCudaKernelParam* buffer, const PxU32 index, T const* first, Ts const*... rest)
