83  PX_CUDA_CALLABLE PX_FORCE_INLINE PxPlane(const PxVec3& p0, const PxVec3& p1, const PxVec3& p2)
84  {

138  {
139  const PxVec3 transformedNormal = pose.rotate(n);
140  return PxPlane(transformedNormal, d - pose.p.dot(transformedNormal));
