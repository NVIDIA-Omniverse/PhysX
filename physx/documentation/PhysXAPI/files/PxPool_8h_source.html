76  // Allocate space for single object
77  PX_INLINE T* allocate()
78  {
79  if(mFreeElement == 0)
80  allocateSlab();
81  T* p = reinterpret_cast<T*>(mFreeElement);
82  mFreeElement = mFreeElement->mNext;
83  mUsed++;
88 #if PX_CHECKED
89  for(uint32_t i = 0; i < sizeof(T); ++i)
90  reinterpret_cast<uint8_t*>(p)[i] = 0xcd;
91 #endif
92  return p;
93  }
94 
95  // Put space for a single element back in the lists
96  PX_INLINE void deallocate(T* p)
97  {
98  if(p)
99  {
100  PX_ASSERT(mUsed);
101  mUsed--;
102  push(reinterpret_cast<FreeList*>(p));
103  }
104  }

184  protected:
185  struct FreeList
186  {
187  FreeList* mNext;
188  };
189 
190  // All the allocated slabs, sorted by pointer
191  PxArray<void*, Alloc> mSlabs;
192 
193  uint32_t mElementsPerSlab;
194  uint32_t mUsed;
195  uint32_t mSlabSize;
196 
197  FreeList* mFreeElement; // Head of free-list
198 
199  // Helper function to get bitmap of allocated elements
200 
201  void push(FreeList* p)
202  {
203  p->mNext = mFreeElement;
204  mFreeElement = p;
205  }
206 
207  // Allocate a slab and segregate it into the freelist
208  void allocateSlab()
209  {
210  T* slab = reinterpret_cast<T*>(Alloc::allocate(mSlabSize, __FILE__, __LINE__));
211 
212  mSlabs.pushBack(slab);
213 
214  // Build a chain of nodes for the freelist
215  T* it = slab + mElementsPerSlab;
216  while(--it >= slab)
217  push(reinterpret_cast<FreeList*>(it));
218  }
219 
220  /*
221  Cleanup method. Go through all active slabs and call destructor for live objects,
222  then free their memory
223  */
224  void disposeElements()
225  {
226  PxArray<void*, Alloc> freeNodes(*this);
227  while(mFreeElement)
228  {
229  freeNodes.pushBack(mFreeElement);
230  mFreeElement = mFreeElement->mNext;
231  }
232  Alloc& alloc(*this);
233  PxSort(freeNodes.begin(), freeNodes.size(), PxLess<void*>(), alloc);
234  PxSort(mSlabs.begin(), mSlabs.size(), PxLess<void*>(), alloc);
235 
236  typename PxArray<void*, Alloc>::Iterator slabIt = mSlabs.begin(), slabEnd = mSlabs.end();
237  for(typename PxArray<void*, Alloc>::Iterator freeIt = freeNodes.begin(); slabIt != slabEnd; ++slabIt)
238  {
239  for(T* tIt = reinterpret_cast<T*>(*slabIt), *tEnd = tIt + mElementsPerSlab; tIt != tEnd; ++tIt)
240  {
241  if(freeIt != freeNodes.end() && *freeIt == tIt)
242  ++freeIt;
243  else
244  tIt->~T();
245  }
246  }
247  }
248 };
