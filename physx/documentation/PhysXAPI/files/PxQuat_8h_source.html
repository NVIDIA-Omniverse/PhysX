159  PX_CUDA_CALLABLE PX_INLINE void toRadiansAndUnitAxis(Type& angle, PxVec3T<Type>& axis) const
160  {
161  const Type quatEpsilon = Type(1.0e-8);
162  const Type s2 = x * x + y * y + z * z;
163  if(s2 < quatEpsilon * quatEpsilon) // can't extract a sensible axis
164  {
165  angle = Type(0.0);
166  axis = PxVec3T<Type>(Type(1.0), Type(0.0), Type(0.0));
167  }
168  else
169  {
170  const Type s = PxRecipSqrt(s2);
171  axis = PxVec3T<Type>(x, y, z) * s;
172  angle = PxAbs(w) < quatEpsilon ? Type(PxPi) : PxAtan2(s2 * s, w) * Type(2.0);
173  }
174  }

287  PX_CUDA_CALLABLE PX_FORCE_INLINE const PxVec3T<Type> rotate(const PxVec3T<Type>& v) const
288  {
289  const Type vx = Type(2.0) * v.x;
290  const Type vy = Type(2.0) * v.y;
291  const Type vz = Type(2.0) * v.z;
292  const Type w2 = w * w - 0.5f;
293  const Type dot2 = (x * vx + y * vy + z * vz);
294  return PxVec3T<Type>((vx * w2 + (y * vz - z * vy) * w + x * dot2), (vy * w2 + (z * vx - x * vz) * w + y * dot2),
295  (vz * w2 + (x * vy - y * vx) * w + z * dot2));
296  }

301  PX_CUDA_CALLABLE PX_FORCE_INLINE const PxVec3T<Type> rotateInv(const PxVec3T<Type>& v) const
302  {
303  const Type vx = Type(2.0) * v.x;
304  const Type vy = Type(2.0) * v.y;
305  const Type vz = Type(2.0) * v.z;
306  const Type w2 = w * w - 0.5f;
307  const Type dot2 = (x * vx + y * vy + z * vz);
308  return PxVec3T<Type>((vx * w2 - (y * vz - z * vy) * w + x * dot2), (vy * w2 - (z * vx - x * vz) * w + y * dot2),
309  (vz * w2 - (x * vy - y * vx) * w + z * dot2));
310  }

324  PX_CUDA_CALLABLE PX_FORCE_INLINE PxQuatT& operator*=(const PxQuatT& q)
325  {
326  const Type tx = w * q.x + q.w * x + y * q.z - q.y * z;
327  const Type ty = w * q.y + q.w * y + z * q.x - q.z * x;
328  const Type tz = w * q.z + q.w * z + x * q.y - q.x * y;
329 
330  w = w * q.w - q.x * x - y * q.y - q.z * z;
331  x = tx;
332  y = ty;
333  z = tz;
334  return *this;
335  }
