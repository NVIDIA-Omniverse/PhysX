170 template<typename HitType>
171 struct PxHitBuffer : public PxHitCallback<HitType>
172 {
183  PxHitBuffer(HitType* aTouches = NULL, PxU32 aMaxNbTouches = 0) : PxHitCallback<HitType>(aTouches, aMaxNbTouches) {}
184 
186  PX_INLINE PxU32 getNbAnyHits() const { return getNbTouches() + PxU32(this->hasBlock); }
188  PX_INLINE const HitType& getAnyHit(const PxU32 index) const { PX_ASSERT(index < getNbTouches() + PxU32(this->hasBlock));
189  return index < getNbTouches() ? getTouches()[index] : this->block; }
190 
191  PX_INLINE PxU32 getNbTouches() const { return this->nbTouches; }
192  PX_INLINE const HitType* getTouches() const { return this->touches; }
193  PX_INLINE const HitType& getTouch(const PxU32 index) const { PX_ASSERT(index < getNbTouches()); return getTouches()[index]; }
194  PX_INLINE PxU32 getMaxNbTouches() const { return this->maxNbTouches; }
195 
196  virtual ~PxHitBuffer() {}
197 
198 protected:
199  // stops after the first callback
200  virtual PxAgain processTouches(const HitType* buffer, PxU32 nbHits) { PX_UNUSED(buffer); PX_UNUSED(nbHits); return false; }
201 };
