223  template<class T>
224  class PxDynamicArrayReportCallback : public PxReportCallback<T>
225  {
226  public:
227  PxDynamicArrayReportCallback(PxArray<T>& results) : mResults(results)
228  {
229  mResults.reserve(32);
230  this->mBuffer = mResults.begin();
231  this->mCapacity = mResults.capacity();
232  }
233 
234  virtual bool flushResults(PxU32 nbItems, const T* /*items*/)
235  {
236  const PxU32 size = mResults.size();
237  const PxU32 capa = mResults.capacity();
238  const PxU32 newSize = size+nbItems;
239  PX_ASSERT(newSize<=capa);
240  mResults.forceSize_Unsafe(newSize);
241  if(newSize==capa)
242  {
243  const PxU32 newCapa = capa*2;
244  mResults.reserve(newCapa);
245  this->mBuffer = mResults.begin() + newSize;
246  this->mCapacity = mResults.capacity() - newSize;
247  }
248  return true;
249  }
250 
251  PxArray<T>& mResults;
252  };
