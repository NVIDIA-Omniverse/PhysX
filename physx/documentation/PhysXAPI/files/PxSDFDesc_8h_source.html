175  PX_INLINE bool PxSDFDesc::isValid() const
176  {
177  // Check validity of user's input(if any)
178  if (sdf.data)
179  {
180  if (dims.x < 1 || dims.y < 1 || dims.z < 1)
181  return false;
182  if (!meshLower.isFinite())
183  return false;
184  if (spacing <= 0)
185  return false;
186  }
187 
188  return true;
189  }
