970 PX_INLINE bool PxSceneDesc::isValid() const
971 {
972  if(!PxSceneQueryDesc::isValid())
973  return false;
974 
975  if(!filterShader)
976  return false;
977 
978  if( ((filterShaderDataSize == 0) && (filterShaderData != NULL)) ||
979  ((filterShaderDataSize > 0) && (filterShaderData == NULL)) )
980  return false;
981 
982  if(!limits.isValid())
983  return false;
984 
985  if(bounceThresholdVelocity <= 0.0f)
986  return false;
987  if(frictionOffsetThreshold < 0.0f)
988  return false;
989  if(frictionCorrelationDistance <= 0)
990  return false;
991 
992  if(maxBiasCoefficient < 0.0f)
993  return false;
994 
995  if(!ccdMaxPasses)
996  return false;
997  if(ccdThreshold <= 0.0f)
998  return false;
999  if(ccdMaxSeparation < 0.0f)
1000  return false;
1001 
1002  if(!cpuDispatcher)
1003  return false;
1004 
1005  if(!contactReportStreamBufferSize)
1006  return false;
1007 
1008  if(maxNbContactDataBlocks < nbContactDataBlocks)
1009  return false;
1010 
1011  if(wakeCounterResetValue <= 0.0f)
1012  return false;
1013 
1014  if(!sanityBounds.isValid())
1015  return false;
1016 
1017 #if PX_SUPPORT_GPU_PHYSX
1018  if(!PxIsPowerOfTwo(gpuMaxNumPartitions))
1019  return false;
1020  if(gpuMaxNumPartitions > 32)
1021  return false;
1022  if (gpuMaxNumPartitions == 0)
1023  return false;
1024  if(!gpuDynamicsConfig.isValid())
1025  return false;
1026 
1027  if (flags & PxSceneFlag::eSUPPRESS_READBACK)
1028  {
1029  if(!(flags & PxSceneFlag::eENABLE_GPU_DYNAMICS && broadPhaseType == PxBroadPhaseType::eGPU))
1030  return false;
1031  }
1032 #endif
1033 
1034  if(contactPairSlabSize == 0)
1035  return false;
1036 
1037  return true;
1038 }
