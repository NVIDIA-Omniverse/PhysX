137 PX_INLINE bool PxSimpleTriangleMesh::isValid() const
138 {
139  // Check geometry
140  if(points.count > 0xffff && flags & PxMeshFlag::e16_BIT_INDICES)
141  return false;
142  if(!points.data)
143  return false;
144  if(points.stride < sizeof(PxVec3)) //should be at least one point's worth of data
145  return false;
146 
147  // Check topology
148  // The triangles pointer is not mandatory
149  if(triangles.data)
150  {
151  // Indexed mesh
152  PxU32 limit = (flags & PxMeshFlag::e16_BIT_INDICES) ? sizeof(PxU16)*3 : sizeof(PxU32)*3;
153  if(triangles.stride < limit)
154  return false;
155  }
156  return true;
157 }
