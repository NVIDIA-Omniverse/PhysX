45 template <class T, class Predicate>
46 PX_INLINE void PxMedian3(T* elements, int32_t first, int32_t last, Predicate& compare)
47 {
48  /*
49  This creates sentinels because we know there is an element at the start minimum(or equal)
50  than the pivot and an element at the end greater(or equal) than the pivot. Plus the
51  median of 3 reduces the chance of degenerate behavour.
52  */
53 
54  int32_t mid = (first + last) / 2;
55 
56  if(compare(elements[mid], elements[first]))
57  PxSwap(elements[first], elements[mid]);
58 
59  if(compare(elements[last], elements[first]))
60  PxSwap(elements[first], elements[last]);
61 
62  if(compare(elements[last], elements[mid]))
63  PxSwap(elements[mid], elements[last]);
64 
65  // keep the pivot at last-1
66  PxSwap(elements[mid], elements[last - 1]);
67 }

69 template <class T, class Predicate>
70 PX_INLINE int32_t PxPartition(T* elements, int32_t first, int32_t last, Predicate& compare)
71 {
72  PxMedian3(elements, first, last, compare);
73 
74  /*
75  WARNING: using the line:
76 
77  T partValue = elements[last-1];
78 
79  and changing the scan loops to:
80 
81  while(comparator.greater(partValue, elements[++i]));
82  while(comparator.greater(elements[--j], partValue);
83 
84  triggers a compiler optimizer bug on xenon where it stores a double to the stack for partValue
85  then loads it as a single...:-(
86  */
87 
88  int32_t i = first; // we know first is less than pivot(but i gets pre incremented)
89  int32_t j = last - 1; // pivot is in last-1 (but j gets pre decremented)
90 
91  for(;;)
92  {
93  while(compare(elements[++i], elements[last - 1]))
94  ;
95  while(compare(elements[last - 1], elements[--j]))
96  ;
97 
98  if(i >= j)
99  break;
100 
101  PX_ASSERT(i <= last && j >= first);
102  PxSwap(elements[i], elements[j]);
103  }
104  // put the pivot in place
105 
106  PX_ASSERT(i <= last && first <= (last - 1));
107  PxSwap(elements[i], elements[last - 1]);
108 
109  return i;
110 }

112 template <class T, class Predicate>
113 PX_INLINE void PxSmallSort(T* elements, int32_t first, int32_t last, Predicate& compare)
114 {
115  // selection sort - could reduce to fsel on 360 with floats.
116 
117  for(int32_t i = first; i < last; i++)
118  {
119  int32_t m = i;
120  for(int32_t j = i + 1; j <= last; j++)
121  if(compare(elements[j], elements[m]))
122  m = j;
123 
124  if(m != i)
125  PxSwap(elements[m], elements[i]);
126  }
127 }

140  {

147 
148  void grow()
149  {
150  mCapacity *= 2;
151  int32_t* newMem =
152  reinterpret_cast<int32_t*>(mAllocator.allocate(sizeof(int32_t) * mCapacity, __FILE__, __LINE__));
153  intrinsics::memCopy(newMem, mMemory, mSize * sizeof(int32_t));
154  if(mRealloc)
155  mAllocator.deallocate(mMemory);
156  mRealloc = true;
157  mMemory = newMem;
158  }
159 
160  PX_INLINE void push(int32_t start, int32_t end)
161  {
162  if(mSize >= mCapacity - 1)
163  grow();
164  mMemory[mSize++] = start;
165  mMemory[mSize++] = end;
166  }
167 
168  PX_INLINE void pop(int32_t& start, int32_t& end)
169  {
170  PX_ASSERT(!empty());
171  end = mMemory[--mSize];
172  start = mMemory[--mSize];
173  }
174 
175  PX_INLINE bool empty()
176  {
177  return mSize == 0;
178  }
179 };
