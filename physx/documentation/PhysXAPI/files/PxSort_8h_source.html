58 template <class T, class Predicate, class PxAllocator>
59 void PxSort(T* elements, uint32_t count, const Predicate& compare, const PxAllocator& inAllocator,
60  const uint32_t initialStackSize = 32)
61 {
62  static const uint32_t SMALL_SORT_CUTOFF = 5; // must be >= 3 since we need 3 for median
63 
64  PX_ALLOCA(stackMem, int32_t, initialStackSize);
65  PxStack<PxAllocator> stack(stackMem, initialStackSize, inAllocator);
66 
67  int32_t first = 0, last = int32_t(count - 1);
68  if(last > first)
69  {
70  for(;;)
71  {
72  while(last > first)
73  {
74  PX_ASSERT(first >= 0 && last < int32_t(count));
75  if(uint32_t(last - first) < SMALL_SORT_CUTOFF)
76  {
77  PxSmallSort(elements, first, last, compare);
78  break;
79  }
80  else
81  {
82  const int32_t partIndex = PxPartition(elements, first, last, compare);
83 
84  // push smaller sublist to minimize stack usage
85  if((partIndex - first) < (last - partIndex))
86  {
87  stack.push(first, partIndex - 1);
88  first = partIndex + 1;
89  }
90  else
91  {
92  stack.push(partIndex + 1, last);
93  last = partIndex - 1;
94  }
95  }
96  }
97 
98  if(stack.empty())
99  break;
100 
101  stack.pop(first, last);
102  }
103  }
104 #if PX_SORT_PARANOIA
105  for(uint32_t i = 1; i < count; i++)
106  PX_ASSERT(!compare(elements[i], elements[i - 1]));
107 #endif
108 }
