95  PX_FORCE_INLINE PxU32 PxCountLeadingZeros(PxU32 v)
96  {
97  PxI32 result = 0;

115  PX_FORCE_INLINE void PxPrefetch(const void* ptr, PxU32 count = 1)
116  {
117  const char* cp = (char*)ptr;
118  PxU64 p = size_t(ptr);
119  PxU64 startLine = p>>6, endLine = (p+count-1)>>6;
120  PxU64 lines = endLine - startLine + 1;
121  do
122  {
123  PxPrefetchLine(cp);
124  cp+=64;
125  } while(--lines);
126  }
