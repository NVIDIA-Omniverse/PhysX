137  PxTetrahedronMeshDesc(physx::PxArray<physx::PxVec3>& meshVertices, physx::PxArray<physx::PxU32>& meshTetIndices, const PxTetrahedronMeshDesc::PxMeshFormat meshFormat = eTET_MESH)

143  tetrahedrons.count = meshTetIndices.size() / 4;
144  tetrahedrons.stride = sizeof(int) * 4;
145  tetrahedrons.data = meshTetIndices.begin();

147  if (meshFormat == eTET_MESH)
148  tetsPerElement = 1;
149  else
150  tetsPerElement = 6;
151  }

153  PX_INLINE bool isValid() const
154  {
155  // Check geometry of the collision mesh
156  if (points.count < 4) //at least 1 tetrahedron's worth of points
157  return false;
158  if ((!tetrahedrons.data) && (points.count % 4)) // Non-indexed mesh => we must ensure the geometry defines an implicit number of tetrahedrons // i.e. numVertices can't be divided by 4
159  return false;
160  if (points.count > 0xffff && flags & PxMeshFlag::e16_BIT_INDICES)
161  return false;
162  if (!points.data)
163  return false;
164  if (points.stride < sizeof(PxVec3)) //should be at least one point's worth of data
165  return false;
166 
167  //add more validity checks here
168  if (materialIndices.data && materialIndices.stride < sizeof(PxFEMMaterialTableIndex))
169  return false;
170 
171  // The tetrahedrons pointer is not mandatory
172  if (tetrahedrons.data)
173  {
174  // Indexed collision mesh
175  PxU32 limit = (flags & PxMeshFlag::e16_BIT_INDICES) ? sizeof(PxU16) * 4 : sizeof(PxU32) * 4;
176  if (tetrahedrons.stride < limit)
177  return false;
178  }
179 
180  //The model can only be either a tetmesh (1 tet per element), or have 5 or 6 tets per hex element, otherwise invalid.
181  if (tetsPerElement != 1 && tetsPerElement != 6)
182  return false;
183 
184  return true;
185  }
186  };
