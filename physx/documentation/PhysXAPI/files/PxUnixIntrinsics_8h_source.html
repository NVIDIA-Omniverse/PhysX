91 PX_FORCE_INLINE void PxPrefetch(const void* ptr, uint32_t count = 1)
92 {
93  const char* cp = reinterpret_cast<const char*>(ptr);
94  uint64_t p = size_t(ptr);
95  uint64_t startLine = p >> 6, endLine = (p + count - 1) >> 6;
96  uint64_t lines = endLine - startLine + 1;
97  do
98  {
99  PxPrefetchLine(cp);
100  cp += 64;
101  } while(--lines);
102 }
