129 PX_CUDA_CALLABLE PX_FORCE_INLINE bool isFinite(float a)
130 {
131  //std::isfinite not recommended as of Feb 2017, since it doesn't work with g++/clang's floating point optimization.
132  union localU { PxU32 i; float f; } floatUnion;
133  floatUnion.f = a;
134  return !((floatUnion.i & 0x7fffffff) >= 0x7f800000);
135 }
