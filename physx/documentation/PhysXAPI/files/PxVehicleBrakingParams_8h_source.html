55 struct PxVehicleBrakeCommandResponseParams : public PxVehicleCommandResponseParams
56 {
57  PX_FORCE_INLINE PxVehicleBrakeCommandResponseParams transformAndScale(
58  const PxVehicleFrame& srcFrame, const PxVehicleFrame& trgFrame, const PxVehicleScale& srcScale, const PxVehicleScale& trgScale) const
59  {
60  PX_UNUSED(srcFrame);
61  PX_UNUSED(trgFrame);
62  PxVehicleBrakeCommandResponseParams r = *this;
63  const PxReal scale = trgScale.scale/srcScale.scale;
64  r.maxResponse *= (scale*scale); //maxResponse is a torque!
65  return r;
66  }

68  PX_FORCE_INLINE bool isValid(const PxVehicleAxleDescription& axleDesc) const
69  {
70  if (!axleDesc.isValid())
71  return false;
72  PX_CHECK_AND_RETURN_VAL(maxResponse >= 0.0f, "PxVehicleBrakeCommandResponseParams.maxResponse must be greater than or equal to zero", false);
73  for (PxU32 i = 0; i < axleDesc.nbWheels; i++)
74  {
75  PX_CHECK_AND_RETURN_VAL(wheelResponseMultipliers[axleDesc.wheelIdsInAxleOrder[i]] >= 0.0f, "PxVehicleBrakeCommandResponseParams.wheelResponseMultipliers[i] must be greater than or equal to zero.", false);
76  }
77  return true;
78  }
