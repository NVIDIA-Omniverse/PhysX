144  //Elements have the form of a linked list to allow traversal over a list of elements.
145  //Each element is either a single component or a subgroup.
146  struct SubGroupElement
147  {
148  SubGroupElement()
149  : childGroup(eINVALID_SUBSTEP_GROUP),
150  component(eINVALID_COMPONENT),

160  //A group is a linked list of elements to be processed in sequence.
161  //Each group stores the first element in the sequence.
162  //Each element in the sequence stores the next element in the sequence
163  //to allow traversal over the list of elements in the group.
164  struct Group
165  {

168  firstElement(eINVALID_SUB_GROUP_ELEMENT),
169  nbSteps(1)
170  {
171  }
172  PxU8 parentGroup;
173  PxU8 firstElement;
174  PxU8 nbSteps;
175  };

188  bool updateSubGroup(const PxReal dt, const PxVehicleSimulationContext& context, const PxU8 groupId, const PxU8 parentSepMultiplier)
189  {
190  const PxU8 nbSteps = mSubGroups[groupId].nbSteps;
191  const PxU8 stepMultiplier = parentSepMultiplier * nbSteps;
192  const PxReal timestepForGroup = dt / PxReal(stepMultiplier);
193  for (PxU8 k = 0; k < nbSteps; k++)
194  {
195  PxU8 nextElement = mSubGroups[groupId].firstElement;
196  while (eINVALID_SUB_GROUP_ELEMENT != nextElement)
197  {
198  const SubGroupElement& e = mSubGroupElements[nextElement];
199  PX_ASSERT(e.component != eINVALID_COMPONENT || e.childGroup != eINVALID_SUBSTEP_GROUP);
200  if (eINVALID_COMPONENT != e.component)
201  {
202  PxVehicleComponent* c = mComponents[e.component];
203  if (!c->update(timestepForGroup, context))
204  return false;
205  }
206  else
207  {
208  PX_ASSERT(eINVALID_SUBSTEP_GROUP != e.childGroup);
209  if (!updateSubGroup(dt, context, e.childGroup, stepMultiplier))
210  return false;
211  }
212  nextElement = e.nextElement;
213  }
214  }
215 
216  return true;
217  }
218 
219  PxU8 getLastKnownElementInGroup(const PxU8 groupId) const
220  {
221  PxU8 currElement = mSubGroups[groupId].firstElement;
222  PxU8 nextElement = mSubGroups[groupId].firstElement;
223  while (nextElement != eINVALID_SUB_GROUP_ELEMENT)
224  {
225  currElement = nextElement;
226  nextElement = mSubGroupElements[nextElement].nextElement;
227  }
228  return currElement;
229  }
230 };

233 bool PxVehicleComponentSequence::add(PxVehicleComponent* c)
234 {
235  if (PxVehicleComponentSequenceLimits::eMAX_NB_COMPONENTS == mNbComponents)
236  return false;
237  if (PxVehicleComponentSequenceLimits::eMAX_NB_SUBGROUPELEMENTS == mNbSubGroupElements)
238  return false;
239 
240  //Create a new element and point it at the component.
241  SubGroupElement& nextElementInGroup = mSubGroupElements[mNbSubGroupElements];
242  nextElementInGroup.childGroup = eINVALID_SUBSTEP_GROUP;
243  nextElementInGroup.component = mNbComponents;
244  nextElementInGroup.nextElement = eINVALID_SUB_GROUP_ELEMENT;
245 
246  if (eINVALID_SUB_GROUP_ELEMENT == mSubGroups[mActiveSubgroup].firstElement)
247  {
248  //The group is empty so add the first element to it.
249  //Point the group at the new element because this will
250  //be the first element in the group.
251  mSubGroups[mActiveSubgroup].firstElement = mNbSubGroupElements;
252  }
253  else
254  {
255  //We are extending the sequence of element of the group.
256  //Add the new element to the end of the group's sequence.
257  mSubGroupElements[getLastKnownElementInGroup(mActiveSubgroup)].nextElement = mNbSubGroupElements;
258  }
259 
260  //Increment the number of elements.
261  mNbSubGroupElements++;
262 
263  //Record the component and increment the number of components.
264  mComponents[mNbComponents] = c;
265  mNbComponents++;
266 
267  return true;
268 }

270 PxU8 PxVehicleComponentSequence::beginSubstepGroup(const PxU8 nbSubSteps)
271 {
272  if (mNbSubgroups == PxVehicleComponentSequenceLimits::eMAX_NB_SUBGROUPS)
273  return eINVALID_SUBSTEP_GROUP;
274  if (mNbSubGroupElements == PxVehicleComponentSequenceLimits::eMAX_NB_SUBGROUPELEMENTS)
275  return eINVALID_SUBSTEP_GROUP;
276 
277  //We have a parent and child group relationship.
278  const PxU8 parentGroup = mActiveSubgroup;
279  const PxU8 childGroup = mNbSubgroups;
280 
281  //Set up the child group.
282  mSubGroups[childGroup].parentGroup = parentGroup;
283  mSubGroups[childGroup].firstElement = eINVALID_SUB_GROUP_ELEMENT;
284  mSubGroups[childGroup].nbSteps = nbSubSteps;
285 
286  //Create a new element to add to the parent group and point it at the child group.
287  SubGroupElement& nextElementIInGroup = mSubGroupElements[mNbSubGroupElements];
288  nextElementIInGroup.childGroup = childGroup;
289  nextElementIInGroup.nextElement = eINVALID_SUB_GROUP_ELEMENT;
290  nextElementIInGroup.component = eINVALID_COMPONENT;
291 
292  //Add the new element to the parent group.
293  if (eINVALID_SUB_GROUP_ELEMENT == mSubGroups[parentGroup].firstElement)
294  {
295  //The parent group is empty so add the first element to it.
296  //Point the parent group at the new element because this will
297  //be the first element in the group.
298  mSubGroups[parentGroup].firstElement = mNbSubGroupElements;
299  }
300  else
301  {
302  //We are extending the sequence of elements of the parent group.
303  //Add the new element to the end of the group's sequence.
304  mSubGroupElements[getLastKnownElementInGroup(parentGroup)].nextElement = mNbSubGroupElements;
305  }
306 
307  //Push the active group.
308  //All subsequent operations will now address the child group and we push or pop the group.
309  mActiveSubgroup = childGroup;
310 
311  //Increment the number of elements.
312  mNbSubGroupElements++;
313 
314  //Increment the number of groups.
315  mNbSubgroups++;
316 
317  //Return the group id.
318  return mActiveSubgroup;
319 }
