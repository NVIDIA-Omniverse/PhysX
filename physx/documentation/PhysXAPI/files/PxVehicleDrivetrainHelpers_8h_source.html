73 PX_FORCE_INLINE PxReal PxVehicleEngineDampingRateCompute
74 (const PxVehicleEngineParams& engineParams,
75  const PxVehicleGearboxParams& gearboxParams, const PxVehicleGearboxState& gearboxState,
76  const PxVehicleClutchCommandResponseState& clutchResponseState,
77  const PxVehicleEngineDriveThrottleCommandResponseState& throttleResponseState)
78 {
79  const PxReal K = (gearboxParams.neutralGear != gearboxState.currentGear) ? clutchResponseState.normalisedCommandResponse : 0.0f;
80  const PxReal zeroThrottleDamping = engineParams.dampingRateZeroThrottleClutchEngaged +
81  (1.0f - K)* (engineParams.dampingRateZeroThrottleClutchDisengaged - engineParams.dampingRateZeroThrottleClutchEngaged);
82  const PxReal appliedThrottle = throttleResponseState.commandResponse;
83  const PxReal fullThrottleDamping = engineParams.dampingRateFullThrottle;
84  const PxReal engineDamping = zeroThrottleDamping + (fullThrottleDamping - zeroThrottleDamping)*appliedThrottle;
85  return engineDamping;
86 }

107 PX_FORCE_INLINE PxReal PxVehicleEngineDriveTorqueCompute
108 (const PxVehicleEngineParams& engineParams, const PxVehicleEngineState& engineState, const PxVehicleEngineDriveThrottleCommandResponseState& throttleCommandResponseState)
109 {
110  const PxReal appliedThrottle = throttleCommandResponseState.commandResponse;
111  const PxReal peakTorque = engineParams.peakTorque;
112  const PxVehicleFixedSizeLookupTable<PxReal, PxVehicleEngineParams::eMAX_NB_ENGINE_TORQUE_CURVE_ENTRIES>& torqueCurve = engineParams.torqueCurve;
113  const PxReal normalisedRotSpeed = engineState.rotationSpeed/engineParams.maxOmega;
114  const PxReal engineDriveTorque = appliedThrottle*peakTorque*torqueCurve.interpolate(normalisedRotSpeed);
115  return engineDriveTorque;
116 }
