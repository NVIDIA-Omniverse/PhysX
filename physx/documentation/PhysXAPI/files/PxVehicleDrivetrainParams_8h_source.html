76  return true;

371  PX_CHECK_AND_RETURN_VAL(nbRatios >= 1, "PxVehicleGearboxParams.nbRatios must be greater than zero", false);
372  PX_CHECK_AND_RETURN_VAL(nbRatios <= eMAX_NB_GEARS, "PxVehicleGearboxParams.nbRatios must be less than or equal to eMAX_NB_GEARS", false);
373  PX_CHECK_AND_RETURN_VAL(neutralGear < nbRatios, "PxVehicleGearboxParams.neutralGear must be less than PxVehicleGearboxParams.nbRatios", false);
374  PX_CHECK_AND_RETURN_VAL(switchTime >= 0.0f, "PxVehicleGearboxParams.switchTime must be greater than or equal to zero", false);
375  for(PxU32 i = 0; i < neutralGear; i++)
376  {
377  PX_CHECK_AND_RETURN_VAL(ratios[i] < 0.0f, "Reverse gear ratios must be less than zero", false);
378  }
379  {
380  PX_CHECK_AND_RETURN_VAL(ratios[neutralGear] == 0.0f, "Neutral gear ratio must be equal to zero", false);
381  }
382  for (PxU32 i = neutralGear + 1; i < nbRatios; i++)
383  {
384  PX_CHECK_AND_RETURN_VAL(ratios[i] > 0.0f, "Forward gear ratios must be greater than zero", false);
385  }
386  for (PxU32 i = neutralGear + 2; i < nbRatios; i++)
387  {
388  PX_CHECK_AND_RETURN_VAL(ratios[i] < ratios[i - 1], "Forward gear ratios must be a descending sequence of gear ratios", false);
389  }
390  return true;
391  }

617  PX_FORCE_INLINE PxVehicleMultiWheelDriveDifferentialParams transformAndScale(
618  const PxVehicleFrame& srcFrame, const PxVehicleFrame& trgFrame, const PxVehicleScale& srcScale, const PxVehicleScale& trgScale) const
619  {
620  PX_UNUSED(srcFrame);

624  return *this;

627  PX_FORCE_INLINE bool isValid(const PxVehicleAxleDescription& axleDesc) const
628  {
629  if (!axleDesc.isValid())
630  return false;
631  PxReal aveWheelSpeedSum = 0.0f;
632  PxReal torqueRatioSum = 0.0f;
633  for (PxU32 i = 0; i < axleDesc.nbWheels; i++)
634  {
635  const PxU32 wheelId = axleDesc.wheelIdsInAxleOrder[i];
636  PX_CHECK_AND_RETURN_VAL(PxAbs(torqueRatios[wheelId]) <= 1.0f, "PxVehicleMultiWheelDriveDifferentialParams.torqueRatios[i] must be in range [-1, 1] for all wheels connected to the differential", false);
637  PX_CHECK_AND_RETURN_VAL(aveWheelSpeedRatios[wheelId] >= 0.0f && aveWheelSpeedRatios[wheelId] <= 1.0f, "PxVehicleMultiWheelDriveDifferentialParams.aveWheelSpeedRatios[i] must be in range [0, 1] for all wheels connected to the differential", false);
638  aveWheelSpeedSum += aveWheelSpeedRatios[wheelId];
639  torqueRatioSum += PxAbs(torqueRatios[wheelId]);
640  }
641  PX_CHECK_AND_RETURN_VAL(aveWheelSpeedSum >= 0.99f && aveWheelSpeedSum <= 1.01f, "Sum of PxVehicleMultiWheelDriveDifferentialParams.aveWheelSpeedRatios[i] must be 1.0.", false);
642  PX_CHECK_AND_RETURN_VAL(torqueRatioSum >= 0.99f && torqueRatioSum <= 1.01f, "Sum of PxVehicleMultiWheelDriveDifferentialParams.torqueRatios[i] must be 1.0.", false);
643  return true;
644  }

749 
750  PX_FORCE_INLINE void setToDefault()
751  {
752  PxVehicleMultiWheelDriveDifferentialParams::setToDefault();
753  frontBias = 0.0f;
754  rearBias = 0.0f;

757 
758  PX_FORCE_INLINE bool isValid(const PxVehicleAxleDescription& axleDesc) const
759  {
760  if (!PxVehicleMultiWheelDriveDifferentialParams::isValid(axleDesc))
761  return false;
762 
763  PX_CHECK_AND_RETURN_VAL((0.0f == centerBias) ||
764  (centerBias > 1.0f &&
765  frontWheelIds[0] < axleDesc.nbWheels && frontWheelIds[1] < axleDesc.nbWheels &&
766  rearWheelIds[0] < axleDesc.nbWheels && rearWheelIds[1] < axleDesc.nbWheels),
767  "PxVehicleFourWheelDriveDifferentialParams:: centreBias is enabled but the frontWheelIds and rearWheelIds are not configured correctly.", false);
768  PX_CHECK_AND_RETURN_VAL((0.0f == centerBias) ||
769  (centerBias > 1.0f &&
770  frontWheelIds[0] != frontWheelIds[1] && frontWheelIds[0] != rearWheelIds[0] && frontWheelIds[0] != rearWheelIds[1] &&
771  frontWheelIds[1] != rearWheelIds[0] && frontWheelIds[1] != rearWheelIds[1] &&
772  rearWheelIds[0] != rearWheelIds[1]),
773  "PxVehicleFourWheelDriveDifferentialParams:: centreBias is enabled but the frontWheelIds and rearWheelIds are not configured correctly.", false);
774  PX_CHECK_AND_RETURN_VAL((0.0f == centerBias) ||
775  (centerBias > 1.0f &&
776  torqueRatios[frontWheelIds[0]] != 0.0f && torqueRatios[frontWheelIds[1]] != 0.0f && torqueRatios[rearWheelIds[0]] != 0.0f && torqueRatios[rearWheelIds[1]] != 0.0f),
777  "PxVehicleFourWheelDriveDifferentialParams:: centreBias is enabled but the front and rear wheels are not connected.", false);
778 
779  PX_CHECK_AND_RETURN_VAL((0.0f == rearBias) ||
780  (rearBias > 1.0f && rearWheelIds[0] < axleDesc.nbWheels && rearWheelIds[1] < axleDesc.nbWheels),
781  "PxVehicleFourWheelDriveDifferentialParams:: rearBias is enabled but the rearWheelIds are not configured correctly.", false);
782  PX_CHECK_AND_RETURN_VAL((0.0f == rearBias) ||
783  (rearBias > 1.0f && rearWheelIds[0] != rearWheelIds[1]),
784  "PxVehicleFourWheelDriveDifferentialParams:: rearBias is enabled but the rearWheelIds are not configured correctly.", false);
785  PX_CHECK_AND_RETURN_VAL((0.0f == centerBias) ||
786  (rearBias > 1.0f && torqueRatios[rearWheelIds[0]] != 0.0f && torqueRatios[rearWheelIds[1]] != 0.0f),
787  "PxVehicleFourWheelDriveDifferentialParams:: rearBias is enabled but the rear wheels are not connected.", false);
788 
789  PX_CHECK_AND_RETURN_VAL((0.0f == frontBias) ||
790  (frontBias > 1.0f && frontWheelIds[0] < axleDesc.nbWheels && frontWheelIds[1] < axleDesc.nbWheels),
791  "PxVehicleFourWheelDriveDifferentialParams:: frontBias is enabled but the frontWheelIds are not configured correctly.", false);
792  PX_CHECK_AND_RETURN_VAL((0.0f == frontBias) ||
793  (frontBias > 1.0f && frontWheelIds[0] != frontWheelIds[1]),
794  "PxVehicleFourWheelDriveDifferentialParams:: frontBias is enabled but the frontWheelIds are not configured correctly.", false);
795  PX_CHECK_AND_RETURN_VAL((0.0f == frontBias) ||
796  (frontBias > 1.0f && torqueRatios[frontWheelIds[0]] != 0.0f && frontWheelIds[frontWheelIds[1]]),
797  "PxVehicleFourWheelDriveDifferentialParams:: frontBias is enabled but the front wheels are not connected.", false);
798 
799  PX_CHECK_AND_RETURN_VAL(((0.0f == frontBias) && (0.0f == rearBias) && (0.0f == centerBias)) || (rate >= 0.0f),
800  "PxVehicleFourWheelDriveDifferentialParams:: strength must be greater than or equal to zero", false);
801 
802  PX_CHECK_AND_RETURN_VAL((0.0f == frontBias) || ((1.0f == frontTarget && 1.0f == rearTarget) || (frontTarget > 1.0f && frontTarget < frontBias)),
803  "PxVehicleFourWheelDriveDifferentialParams: frontBias is enabled but frontTarget not in range (1.0f, frontBias)", false);
804  PX_CHECK_AND_RETURN_VAL((0.0f == rearBias) || ((1.0f == rearTarget && 1.0f == rearBias) || (rearTarget > 1.0f && rearTarget < rearBias)),
805  "PxVehicleFourWheelDriveDifferentialParams: rearBias is enabled but rearTarget not in range (1.0f, rearBias)", false);
806  PX_CHECK_AND_RETURN_VAL((0.0f == centerBias) || ((1.0f == centerTarget && 1.0f == centerBias) || (centerTarget > 1.0f && centerTarget < centerBias)),
807  "PxVehicleFourWheelDriveDifferentialParams: centerBias is enabled but centerTarget not in range (1.0f, centerBias)", false);
808 
809  return true;
810  }
