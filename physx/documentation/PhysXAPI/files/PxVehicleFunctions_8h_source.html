50 PX_FORCE_INLINE PxVec3 PxVehicleTransformFrameToFrame
51 (const PxVehicleFrame& srcFrame, const PxVehicleFrame& trgFrame, const PxVec3& v)
52 {
53  PxVec3 result = v;
54  if ((srcFrame.lngAxis != trgFrame.lngAxis) || (srcFrame.latAxis != trgFrame.latAxis) || (srcFrame.vrtAxis != trgFrame.vrtAxis))
55  {
56  const PxMat33 a = srcFrame.getFrame();
57  const PxMat33 r = trgFrame.getFrame();
58  result = (r * a.getTranspose() * v);
59  }
60  return result;
61 }

63 PX_FORCE_INLINE PxVec3 PxVehicleTransformFrameToFrame
64 (const PxVehicleFrame& srcFrame, const PxVehicleFrame& trgFrame,
65  const PxVehicleScale& srcScale, const PxVehicleScale& trgScale,
66  const PxVec3& v)
67 {
68  PxVec3 result = PxVehicleTransformFrameToFrame(srcFrame, trgFrame, v);
69  if((srcScale.scale != trgScale.scale))
70  result *= (trgScale.scale / srcScale.scale);
71  return result;
72 }

74 PX_FORCE_INLINE PxTransform PxVehicleTransformFrameToFrame
75 (const PxVehicleFrame& srcFrame, const PxVehicleFrame& trgFrame,
76  const PxVehicleScale& srcScale, const PxVehicleScale& trgScale,
77  const PxTransform& v)
78 {
79  PxTransform result(PxVehicleTransformFrameToFrame(srcFrame, trgFrame, srcScale, trgScale, v.p), v.q);
80  if ((srcFrame.lngAxis != trgFrame.lngAxis) || (srcFrame.latAxis != trgFrame.latAxis) || (srcFrame.vrtAxis != trgFrame.vrtAxis))
81  {
82  PxF32 angle;
83  PxVec3 axis;
84  v.q.toRadiansAndUnitAxis(angle, axis);
85  result.q = PxQuat(angle, PxVehicleTransformFrameToFrame(srcFrame, trgFrame, axis));
86  }
87  return result;
88 }

96 PX_FORCE_INLINE PxQuat PxVehicleComputeRotation(const PxVehicleFrame& frame, const PxReal roll, const PxReal pitch, const PxReal yaw)
97 {
98  const PxMat33 m = frame.getFrame();
99  const PxVec3& lngAxis = m.column0;
100  const PxVec3& latAxis = m.column1;
101  const PxVec3& vrtAxis = m.column2;
102  const PxQuat quatPitch(pitch, latAxis);
103  const PxQuat quatRoll(roll, lngAxis);
104  const PxQuat quatYaw(yaw, vrtAxis);
105  const PxQuat result = quatYaw * quatRoll * quatPitch;
106  return result;
107 }

135 PX_FORCE_INLINE void PxVehicleShiftOrigin
136 (const PxVehicleAxleDescription& axleDesc, const PxVec3& shift,
137  PxVehicleRigidBodyState& rigidBodyState, PxVehicleRoadGeometryState* roadGeometryStates,
138  PxVehiclePhysXActor* physxActor = NULL, PxVehiclePhysXRoadGeometryQueryState* physxQueryStates = NULL)
139 {
140  //Adjust the vehicle's internal pose.
141  rigidBodyState.pose.p -= shift;
142 
143  //Optionally adjust the PxRigidActor pose.
144  if (physxActor && !physxActor->rigidBody->getScene())
145  {
146  const PxTransform oldPose = physxActor->rigidBody->getGlobalPose();
147  const PxTransform newPose(oldPose.p - shift, oldPose.q);
148  physxActor->rigidBody->setGlobalPose(newPose);
149  }
150 
151  for (PxU32 i = 0; i < axleDesc.nbWheels; i++)
152  {
153  const PxU32 wheelId = axleDesc.wheelIdsInAxleOrder[i];
154 
155  //Optionally adjust the hit position.
156  if (physxQueryStates && physxQueryStates[wheelId].actor)
157  physxQueryStates[wheelId].hitPosition -= shift;
158 
159  //Adjust the hit plane.
160  if (roadGeometryStates[wheelId].hitState)
161  {
162  const PxPlane plane = roadGeometryStates[wheelId].plane;
163  PxU32 largestNormalComponentAxis = 0;
164  PxReal largestNormalComponent = 0.0f;
165  const PxF32 normalComponents[3] = { plane.n.x, plane.n.y, plane.n.z };
166  for (PxU32 k = 0; k < 3; k++)
167  {
168  if (PxAbs(normalComponents[k]) > largestNormalComponent)
169  {
170  largestNormalComponent = PxAbs(normalComponents[k]);
171  largestNormalComponentAxis = k;
172  }
173  }
174  PxVec3 pointInPlane(PxZero);
175  switch (largestNormalComponentAxis)
176  {
177  case 0:
178  pointInPlane.x = -plane.d / plane.n.x;
179  break;
180  case 1:
181  pointInPlane.y = -plane.d / plane.n.y;
182  break;
183  case 2:
184  pointInPlane.z = -plane.d / plane.n.z;
185  break;
186  default:
187  break;
188  }
189  roadGeometryStates[wheelId].plane.d = -plane.n.dot(pointInPlane - shift);
190  }
191  }
192 }
