104  wheelParams, suspensionParams, materialFrictionParams,

106 
107  if (context.getType() == PxVehicleSimulationContextType::ePHYSX)

111  for(PxU32 i = 0; i < axleDescription->nbWheels; i++)

118  steerResponseStates[wheelId], *rigidBodyState,
119  *physxContext.physxScene, physxContext.physxUnitCylinderSweepMesh, context.frame,
120  roadGeometryStates[wheelId],
121  !physxRoadGeometryStates.isEmpty() ? &physxRoadGeometryStates[wheelId] : NULL);

131 
132  roadGeometryStates[wheelId].setToDefault();
133  }
134  }
135 
136  return true;
137  }
