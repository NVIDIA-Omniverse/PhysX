368 PX_INLINE bool PxVehicleContext::isValid() const
369 {
370  if (!upAxis.isNormalized())
371  return false;
372 
373  if (!forwardAxis.isNormalized())
374  return false;
375 
376  if (!sideAxis.isNormalized())
377  return false;
378 
379  if (((upAxis.cross(forwardAxis)) - sideAxis).magnitude() > 0.02f) // somewhat above 1 degree assuming both have unit length
380  return false;
381 
382  if ((pointRejectAngleThresholdCosine >= 1) || (pointRejectAngleThresholdCosine <= -1))
383  return false;
384 
385  if ((normalRejectAngleThresholdCosine >= 1) || (normalRejectAngleThresholdCosine <= -1))
386  return false;
387 
388  if (maxHitActorAcceleration < 0.0f)
389  return false;
390 
391  return true;
392 }
