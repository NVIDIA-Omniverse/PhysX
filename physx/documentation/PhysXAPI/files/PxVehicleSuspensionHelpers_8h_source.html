69 PX_FORCE_INLINE PxTransform PxVehicleComputeWheelPoseForSuspensionQuery(const PxVehicleFrame& frame, const PxVehicleSuspensionParams& suspensionParams,
70  const PxReal steerAngle, const PxTransform& rigidBodyPose)
71 {
72  //Compute the wheel pose with zero travel from attachment point, zero compliance,
73  //zero wheel pitch (ignore due to radial symmetry).
74  //Zero travel from attachment point (we want the wheel pose at the top of the suspension, i.e., at max compression)
75  PxVehicleSuspensionState suspState;
76  suspState.setToDefault();
77  //Compute the wheel pose.
78  const PxTransform wheelPose = PxVehicleComputeWheelPose(frame, suspensionParams, suspState, 0.0f, 0.0f, steerAngle, rigidBodyPose,
79  0.0f);
80 
81  return wheelPose;
82 }

97 PX_FORCE_INLINE void PxVehicleComputeSuspensionRaycast
98 (const PxVehicleFrame& frame, const PxVehicleWheelParams& wheelParams, const PxVehicleSuspensionParams& suspensionParams,
99  const PxF32 steerAngle, const PxTransform& rigidBodyPose,
100  PxVec3& start, PxVec3& dir, PxReal& dist)
101 {
102  const PxTransform wheelPose = PxVehicleComputeWheelPoseForSuspensionQuery(frame, suspensionParams, steerAngle, rigidBodyPose);
103 
104  //Raycast from top of wheel at max compression to bottom of wheel at max droop.
105  dir = PxVehicleComputeSuspensionDirection(suspensionParams, rigidBodyPose);
106  start = wheelPose.p - dir * wheelParams.radius;
107  dist = suspensionParams.suspensionTravelDist + 2.0f*wheelParams.radius;
108 }
