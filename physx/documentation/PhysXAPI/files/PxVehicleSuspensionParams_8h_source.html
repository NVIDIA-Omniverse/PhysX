80  PX_FORCE_INLINE PxVehicleSuspensionParams transformAndScale(
81  const PxVehicleFrame& srcFrame, const PxVehicleFrame& trgFrame, const PxVehicleScale& srcScale, const PxVehicleScale& trgScale) const
82  {

154 struct PxVehicleSuspensionComplianceParams
155 {
160  PxVehicleFixedSizeLookupTable<PxReal, 3> wheelToeAngle;
161 
166  PxVehicleFixedSizeLookupTable<PxReal, 3> wheelCamberAngle;
167 
172  PxVehicleFixedSizeLookupTable<PxVec3, 3> suspForceAppPoint;
173 
178  PxVehicleFixedSizeLookupTable<PxVec3, 3> tireForceAppPoint;
179 
180  PX_FORCE_INLINE PxVehicleSuspensionComplianceParams transformAndScale(
181  const PxVehicleFrame& srcFrame, const PxVehicleFrame& trgFrame, const PxVehicleScale& srcScale, const PxVehicleScale& trgScale) const
182  {
183  PX_UNUSED(srcFrame);
184  PX_UNUSED(trgFrame);
185  PxVehicleSuspensionComplianceParams r = *this;
186 
187  const PxReal scale = trgScale.scale / srcScale.scale;
188  for (PxU32 i = 0; i < r.suspForceAppPoint.nbDataPairs; i++)
189  {
190  r.suspForceAppPoint.yVals[i] = PxVehicleTransformFrameToFrame(srcFrame, trgFrame, suspForceAppPoint.yVals[i]);
191  r.suspForceAppPoint.yVals[i] *= scale;
192  }
193  for (PxU32 i = 0; i < r.tireForceAppPoint.nbDataPairs; i++)
194  {
195  r.tireForceAppPoint.yVals[i] = PxVehicleTransformFrameToFrame(srcFrame, trgFrame, tireForceAppPoint.yVals[i]);
196  r.tireForceAppPoint.yVals[i] *= scale;
197  }
198  return r;
199  }
200 
201  PX_FORCE_INLINE bool isValid() const
202  {
203  PX_CHECK_AND_RETURN_VAL(wheelToeAngle.isValid(), "PxVehicleSuspensionComplianceParams.wheelToeAngle is invalid", false);
204  PX_CHECK_AND_RETURN_VAL(wheelCamberAngle.isValid(), "PxVehicleSuspensionComplianceParams.wheelCamberAngle is invalid", false);
205  PX_CHECK_AND_RETURN_VAL(suspForceAppPoint.isValid(), "PxVehicleSuspensionComplianceParams.wheelToeAngle is invalid", false);
206  PX_CHECK_AND_RETURN_VAL(tireForceAppPoint.isValid(), "PxVehicleSuspensionComplianceParams.wheelCamberAngle is invalid", false);
207 
208  for (PxU32 i = 0; i < wheelToeAngle.nbDataPairs; i++)
209  {
210  PX_CHECK_AND_RETURN_VAL(wheelToeAngle.xVals[i] >= 0.0f && wheelToeAngle.xVals[i] <= 1.0f, "PxVehicleSuspensionComplianceParams.wheelToeAngle must be an array of points (x,y) with x in range [0, 1]", false);
211  PX_CHECK_AND_RETURN_VAL(wheelToeAngle.yVals[i] >= -PxPi && wheelToeAngle.yVals[i] <= PxPi, "PxVehicleSuspensionComplianceParams.wheelToeAngle must be an array of points (x,y) with y in range [-Pi, Pi]", false);
212  }
213  for (PxU32 i = 0; i < wheelCamberAngle.nbDataPairs; i++)
214  {
215  PX_CHECK_AND_RETURN_VAL(wheelCamberAngle.xVals[i] >= 0.0f && wheelCamberAngle.xVals[i] <= 1.0f, "PxVehicleSuspensionComplianceParams.wheelCamberAngle must be an array of points (x,y) with x in range [0, 1]", false);
216  PX_CHECK_AND_RETURN_VAL(wheelCamberAngle.yVals[i] >= -PxPi && wheelCamberAngle.yVals[i] <= PxPi, "PxVehicleSuspensionComplianceParams.wheelCamberAngle must be an array of points (x,y) with y in range [-Pi, Pi]", false);
217  }
218 
219  for (PxU32 i = 0; i < suspForceAppPoint.nbDataPairs; i++)
220  {
221  PX_CHECK_AND_RETURN_VAL(suspForceAppPoint.xVals[i] >= 0.0f && suspForceAppPoint.xVals[i] <= 1.0f, "PxVehicleSuspensionComplianceParams.suspForceAppPoint[0] must be an array of points (x,y) with x in range [0, 1]", false);
222  }
223  for (PxU32 i = 0; i < tireForceAppPoint.nbDataPairs; i++)
224  {
225  PX_CHECK_AND_RETURN_VAL(tireForceAppPoint.xVals[i] >= 0.0f && tireForceAppPoint.xVals[i] <= 1.0f, "PxVehicleSuspensionComplianceParams.tireForceAppPoint[0] must be an array of points (x,y) with x in range [0, 1]", false);
226  }
227  return true;
228  }

345 struct PxVehicleAntiRollForceParams
346 {
347  /*
348  \brief The anti-roll bar connects two wheels with indices wheel0 and wheel1
349  \note wheel0 and wheel1 may be chosen to have the effect of an anti-dive bar or to have the effect of an anti-roll bar.
350  */
351  PxU32 wheel0;
352 
353  /*
354  \brief The anti-roll bar connects two wheels with indices wheel0 and wheel1
355  \note wheel0 and wheel1 may be chosen to have the effect of an anti-dive bar or to have the effect of an anti-roll bar.
356  */
357  PxU32 wheel1;
358 
359  /*
360  \brief The linear stiffness of the anti-roll bar.
361  \note A positive stiffness will work to reduce the discrepancy in jounce between wheel0 and wheel1.
362  \note A negative stiffness will work to increase the discrepancy in jounce between wheel0 and wheel1.
363 
364  <b>Unit:</b> mass / (time^2)
365  */
366  PxReal stiffness;
367 
368  PX_FORCE_INLINE PxVehicleAntiRollForceParams transformAndScale(
369  const PxVehicleFrame& srcFrame, const PxVehicleFrame& trgFrame, const PxVehicleScale& srcScale, const PxVehicleScale& trgScale) const
370  {
371  PX_UNUSED(srcFrame);
372  PX_UNUSED(trgFrame);
373  PX_UNUSED(srcScale);
374  PX_UNUSED(trgScale);
375  return *this;
376  }
377 
378  PX_FORCE_INLINE bool isValid(const PxVehicleAxleDescription& axleDesc) const
379  {
380  if (!PxIsFinite(stiffness))
381  return false;
382  if (wheel0 == wheel1)
383  return false;
384 
385  //Check that each wheel id is a valid wheel.
386  const PxU32 wheelIds[2] = { wheel0, wheel1 };
387  for (PxU32 k = 0; k < 2; k++)
388  {
389  const PxU32 wheelToFind = wheelIds[k];
390  bool foundWheelInAxleDescription = false;
391  for (PxU32 i = 0; i < axleDesc.nbWheels; i++)
392  {
393  const PxU32 wheel = axleDesc.wheelIdsInAxleOrder[i];
394  if (wheel == wheelToFind)
395  {
396  foundWheelInAxleDescription = true;
397  }
398  }
399  if (!foundWheelInAxleDescription)
400  return false;
401  }
402 
403  return true;
404  }
405 };
