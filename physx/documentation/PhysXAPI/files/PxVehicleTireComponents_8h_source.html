84  virtual bool update(const PxReal dt, const PxVehicleSimulationContext& context)
85  {
86  PX_PROFILE_ZONE("PxVehicleTireComponent::update", 0);
87 
88  const PxVehicleAxleDescription* axleDescription;
89  PxVehicleArrayData<const PxReal> steerResponseStates;
90  const PxVehicleRigidBodyState* rigidBodyState;
91  PxVehicleArrayData<const PxVehicleWheelActuationState> actuationStates;
92  PxVehicleArrayData<const PxVehicleWheelParams> wheelParams;
93  PxVehicleArrayData<const PxVehicleSuspensionParams> suspensionParams;
94  PxVehicleArrayData<const PxVehicleTireForceParams> tireForceParams;
95  PxVehicleArrayData<const PxVehicleRoadGeometryState> roadGeomStates;
96  PxVehicleArrayData<const PxVehicleSuspensionState> suspensionStates;
97  PxVehicleArrayData<const PxVehicleSuspensionComplianceState> suspensionComplianceStates;
98  PxVehicleArrayData<const PxVehicleSuspensionForce> suspensionForces;
99  PxVehicleArrayData<const PxVehicleWheelRigidBody1dState> wheelRigidBody1DStates;
100  PxVehicleArrayData<PxVehicleTireGripState> tireGripStates;
101  PxVehicleArrayData<PxVehicleTireDirectionState> tireDirectionStates;
102  PxVehicleArrayData<PxVehicleTireSpeedState> tireSpeedStates;
103  PxVehicleArrayData<PxVehicleTireSlipState> tireSlipStates;
104  PxVehicleArrayData<PxVehicleTireCamberAngleState> tireCamberAngleStates;
105  PxVehicleArrayData<PxVehicleTireStickyState> tireStickyStates;
106  PxVehicleArrayData<PxVehicleTireForce> tireForces;
107 
108  getDataForTireComponent(axleDescription, steerResponseStates,
109  rigidBodyState, actuationStates, wheelParams, suspensionParams, tireForceParams,
110  roadGeomStates, suspensionStates, suspensionComplianceStates, suspensionForces,
111  wheelRigidBody1DStates, tireGripStates, tireDirectionStates, tireSpeedStates,
112  tireSlipStates, tireCamberAngleStates, tireStickyStates, tireForces);
113 
114  for (PxU32 i = 0; i < axleDescription->nbWheels; i++)
115  {
116  const PxU32 wheelId = axleDescription->wheelIdsInAxleOrder[i];
117 
118  //Compute the tire slip directions
119  PxVehicleTireDirsUpdate(
120  suspensionParams[wheelId],
121  steerResponseStates[wheelId],
122  roadGeomStates[wheelId], suspensionComplianceStates[wheelId],
123  *rigidBodyState,
124  context.frame,
125  tireDirectionStates[wheelId]);
126 
127  //Compute the rigid body speeds along the tire slip directions.
128  PxVehicleTireSlipSpeedsUpdate(
129  wheelParams[wheelId], suspensionParams[wheelId],
130  steerResponseStates[wheelId], suspensionStates[wheelId], tireDirectionStates[wheelId],
131  *rigidBodyState, roadGeomStates[i],
132  context.frame,
133  tireSpeedStates[wheelId]);
134 
135  //Compute the tire slip angles.
136  PxVehicleTireSlipsUpdate(
137  wheelParams[wheelId], context.tireSlipParams,
138  actuationStates[wheelId], tireSpeedStates[wheelId],
139  wheelRigidBody1DStates[wheelId],
140  tireSlipStates[wheelId]);
141 
142  //Update the camber angle
143  PxVehicleTireCamberAnglesUpdate(
144  suspensionParams[wheelId], steerResponseStates[wheelId], roadGeomStates[wheelId],
145  suspensionComplianceStates[wheelId], *rigidBodyState,
146  context.frame,
147  tireCamberAngleStates[wheelId]);
148 
149  //Compute the friction
150  PxVehicleTireGripUpdate(
151  tireForceParams[wheelId], roadGeomStates[wheelId],
152  suspensionStates[wheelId], suspensionForces[wheelId],
153  tireSlipStates[wheelId], tireGripStates[wheelId]);
154 
155  //Update the tire sticky state
156  //
157  //Note: this should be skipped if tires do not use the sticky feature
158  PxVehicleTireStickyStateUpdate(
159  *axleDescription,
160  wheelParams[wheelId],
161  context.tireStickyParams,
162  actuationStates, tireGripStates[wheelId],
163  tireSpeedStates[wheelId], wheelRigidBody1DStates[wheelId],
164  dt,
165  tireStickyStates[wheelId]);
166 
167  //If sticky tire is active set the slip angle to zero.
168  //
169  //Note: this should be skipped if tires do not use the sticky feature
170  PxVehicleTireSlipsAccountingForStickyStatesUpdate(
171  tireStickyStates[wheelId],
172  tireSlipStates[wheelId]);
173 
174  //Compute the tire forces
175  PxVehicleTireForcesUpdate(
176  wheelParams[wheelId], suspensionParams[wheelId],
177  tireForceParams[wheelId],
178  suspensionComplianceStates[wheelId],
179  tireGripStates[wheelId], tireDirectionStates[wheelId],
180  tireSlipStates[wheelId], tireCamberAngleStates[wheelId],
181  *rigidBodyState,
182  tireForces[wheelId]);
183  }
184 
185  return true;
186  }
187 };
