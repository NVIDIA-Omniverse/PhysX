32 PX_INLINE physx::PxSIMDGuard::PxSIMDGuard(bool enable) : mEnabled(enable)
33 {
34 #if !PX_ARM && !PX_A64
35  if (enable)
36  {
37  mControlWord = _mm_getcsr();
38  // set default (disable exceptions: _MM_MASK_MASK) and FTZ (_MM_FLUSH_ZERO_ON), DAZ (_MM_DENORMALS_ZERO_ON: (1<<6))
39  _mm_setcsr(_MM_MASK_MASK | _MM_FLUSH_ZERO_ON | (1 << 6));
40  }
41  else
42  {
43  PX_ASSERT(_mm_getcsr() & _MM_FLUSH_ZERO_ON);
44  PX_ASSERT(_mm_getcsr() & (1 << 6));
45  PX_ASSERT(_mm_getcsr() & _MM_MASK_MASK);
46  }
47 #endif
48 }

50 PX_INLINE physx::PxSIMDGuard::~PxSIMDGuard()
51 {
52 #if !PX_ARM && !PX_A64
53  if (mEnabled)
54  {
55  // restore control word and clear any exception flags
56  // (setting exception state flags cause exceptions on the first following fp operation)
57  _mm_setcsr(mControlWord & ~_MM_EXCEPT_MASK);
58  }
59 #endif
60 }
