104 PX_FORCE_INLINE uint32_t PxCountLeadingZeros(uint32_t v)
105 {
106  if(v)

120 PX_FORCE_INLINE void PxPrefetchLine(const void* ptr, uint32_t offset = 0)
121 {
122  // cache line on X86/X64 is 64-bytes so a 128-byte prefetch would require 2 prefetches.
123  // However, we can only dispatch a limited number of prefetch instructions so we opt to prefetch just 1 cache line
124  /*_mm_prefetch(((const char*)ptr + offset), _MM_HINT_T0);*/
125  // We get slightly better performance prefetching to non-temporal addresses instead of all cache levels
126  _mm_prefetch(((const char*)ptr + offset), _MM_HINT_NTA);
127 }
128 #else
129 PX_FORCE_INLINE void PxPrefetchLine(const void* ptr, uint32_t offset = 0)
130 {
131  // arm does have 32b cache line size
132  __prefetch(((const char*)ptr + offset));
133 }
134 #endif

139 #if !PX_ARM
140 PX_FORCE_INLINE void PxPrefetch(const void* ptr, uint32_t count = 1)
141 {
142  const char* cp = (char*)ptr;
143  uint64_t p = size_t(ptr);
144  uint64_t startLine = p >> 6, endLine = (p + count - 1) >> 6;
145  uint64_t lines = endLine - startLine + 1;
146  do
147  {
148  PxPrefetchLine(cp);
149  cp += 64;
150  } while(--lines);
151 }
